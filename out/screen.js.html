document.addEventListener('DOMContentLoaded', () => {
    const enterScreen = document.getElementById('enter_screen');
    const startScreen = document.getElementById('start_screen');
    const loadingScreen = document.getElementById('loading_screen');
    const canvas = document.getElementById('canvas');
    const showSettingbutton = document.getElementById('overlay-handle');
    const introTheme = new Audio('audio/intro_theme.mp3');
    introTheme.loop = true;
    soundManager.registerMusic(introTheme);

    enterScreen.style.display = 'flex';

    /**
  * Handles entering the start screen when the Enter key is pressed or a click occurs on the enter screen.
  * @param {Event} event - The key press or click event.
  */
    const handleEnterScreen = (event) => {
        if (event.type === 'keydown' && event.key !== 'Enter') return;
        if (event.type === 'click' && event.target !== enterScreen) return;

        introTheme.play().catch((error) => console.error('Error playing intro music:', error));
        enterScreen.style.display = 'none';
        startScreen.style.display = 'flex';

        document.removeEventListener('keydown', handleEnterScreen);
        document.removeEventListener('click', handleEnterScreen);
    };

    document.addEventListener('keydown', handleEnterScreen);
    document.addEventListener('click', handleEnterScreen);


    const playIcon = document.getElementById('play-icon');
    playIcon.addEventListener('click', () => {
        startScreen.style.display = 'none';
        loadingScreen.style.display = 'flex';

        setTimeout(() => {
            loadingScreen.style.display = 'none';
            canvas.style.display = 'block';
            showSettingbutton.style.display = 'flex';
            introTheme.pause();
            introTheme.currentTime = 0;
            isGameStarted = true;
            init();
        }, 2000);
    });

    const exitIcon = document.getElementById('exit-icon');
    exitIcon.addEventListener('click', () => {
        // location.reload();
    });
});

    const exitCancel = document.getElementById('exit-cancel');

    if (!exitIcon || !exitMenu || !exitConfirm || !exitCancel) {
        console.error("‚ùå Exit-Overlay-Elemente nicht gefunden!");
        return;
    }

    /**
     * √ñffnet das Exit-Overlay.
     */
    exitIcon.addEventListener('click', () => {
        exitMenu.style.display = 'block';
    });

    /**
     * Schlie√üt das Exit-Overlay ohne das Spiel zu beenden.
     */
    exitCancel.addEventListener('click', () => {
        exitMenu.style.display = 'none';
    });

    /**
     * **Beendet das Spiel nur nach Best√§tigung (`exit-confirm`).**
     */
    exitConfirm.addEventListener('click', () => {
        exitMenu.style.display = 'none';

        if (window.electron && window.electron.exitApp) {
            console.log("‚ö° Electron erkannt ‚Üí App wird geschlossen...");
            window.electron.exitApp(); // Electron: App beenden
        } else {
            exitGame(); // Normaler Browser: Fenster schlie√üen oder Fallback
        }
    });

    /**
     * Versucht das Spiel zu beenden (Browser).
     * Falls `window.close()` blockiert wird, wird als Fallback `about:blank` ge√∂ffnet.
     */
    function exitGame() {
        console.log("üõë Versuche das Spiel zu beenden...");

        try {
            window.close();

            setTimeout(() => {
                if (!window.closed) {
                    console.warn("‚ùå Fenster konnte nicht automatisch geschlossen werden.");
                    redirectToBlank();
                }
            }, 500);
        } catch (error) {
            console.error("‚ùå Fehler beim Beenden des Spiels:", error);
            redirectToBlank();
        }
    }

    /**
     * **√ñffnet `about:blank`, falls `window.close()` blockiert wird.**
     */
    function redirectToBlank() {
        let newTab = window.open("about:blank", "_blank");

        if (newTab) {
            newTab.document.write(`
                <html>
                <head>
                    <title>Game Closed</title>

/**
 * Checks if the "showEnterScreen" parameter exists in the URL.
 * Displays the enter screen if the value is set to "true".

function checkForEnterScreen() {
    const urlParams = new URLSearchParams(window.location.search);
    if (urlParams.get('showEnterScreen') === 'true') {
        document.getElementById('enter_screen').style.display = 'block';
        document.getElementById('start_screen').style.display = 'none';
        document.getElementById('game_container').style.display = 'none';

    }
}

document.addEventListener("DOMContentLoaded", () => {
    const canvas = document.getElementById("canvas");
    const fullscreenButton = document.getElementById("fullscreen-toggle");
    const exitButton = document.getElementById("exit-fullscreen");
    const settingsButton = document.getElementById("settings-button");
    const settingsMenu = document.getElementById("settings-menu");
    const touchControls = document.getElementById("touch-controls");
    const controls = document.querySelector(".controls");
    const showControlsButton = document.getElementById("show-control-buttons");

    let controlsVisible = false; // Standardm√§√üig ausgeblendet

    if (!canvas) {
        console.error(" Canvas nicht gefunden!");
        return;
    }

    /**
     * Steuerung-Buttons ein-/ausblenden.
     */
    showControlsButton.addEventListener("click", () => {
        controlsVisible = !controlsVisible; // Zustand umkehren

        if (controlsVisible) {
            controls.style.display = "flex";
            touchControls.style.display = "flex";
            console.log("üéÆ Steuerung sichtbar!");
        } else {
            controls.style.display = "none";
            touchControls.style.display = "none";
            console.log("üéÆ Steuerung versteckt!");
        }
    });

    /**
     * Aktiviert oder deaktiviert den Fullscreen-Modus.
     */
    function toggleFullscreen() {
        const canvas = document.getElementById("canvas"); // Falls n√∂tig, ID anpassen
        const topOffset = 170;  // Platz f√ºr oberes Overlay
        const bottomOffset = 170;  // Platz f√ºr unteres Overlay

        // Pr√ºfe, ob die Bedingung f√ºr Fullscreen-Verbot zutrifft
        let screenWidth = window.innerWidth;
        let screenHeight = window.innerHeight;
        let isRestrictedSize = screenWidth >= 600 && screenHeight <= 699;

        console.log("üìè Bildschirmbreite:", screenWidth);
        console.log("üìè Bildschirmh√∂he:", screenHeight);
        console.log("‚õî Fullscreen erlaubt?", !isRestrictedSize);

        if (isRestrictedSize) {
            console.warn("‚ö† Fullscreen-Modus ist auf dieser Aufl√∂sung nicht erlaubt!");
            if (document.fullscreenElement) {
                exitFullscreen(); // Falls Fullscreen aktiv ist, sofort deaktivieren
            }
            return; // Funktion verlassen, um Fullscreen zu blockieren
        }

        if (!document.fullscreenElement) {
            document.documentElement.requestFullscreen().then(() => {
                canvas.style.width = "100vw"; // Volle Breite nutzen
                canvas.style.height = `calc(100vh - ${topOffset + bottomOffset}px)`; // H√∂he exakt berechnen
                canvas.style.position = "absolute";
                canvas.style.top = `${topOffset}px`; // Oben b√ºndig setzen
                canvas.style.bottom = `${bottomOffset}px`; // Unten exakt setzen
                canvas.style.left = "0";
                canvas.style.transform = "none";

                ensureControlsVisible(); // Buttons sichtbar halten
                console.log("‚úÖ Fullscreen aktiviert");
            }).catch(err => console.error("‚ùå Fehler beim Aktivieren von Fullscreen:", err));
        } else {
            exitFullscreen();
        }
    }

    // Funktion zum Beenden des Fullscreen-Modus
    function exitFullscreen() {
        if (document.fullscreenElement) {
            document.exitFullscreen().then(() => {
                console.log("üì¥ Fullscreen deaktiviert");
            }).catch(err => console.error("‚ùå Fehler beim Deaktivieren von Fullscreen:", err));
        }
    }

    // √úberwache Gr√∂√üen√§nderungen und deaktiviere Fullscreen, wenn n√∂tig
    window.addEventListener("resize", () => {
        let screenWidth = window.innerWidth;
        let screenHeight = window.innerHeight;
        let isRestrictedSize = screenWidth >= 600 && screenHeight <= 699;

        if (isRestrictedSize && document.fullscreenElement) {
            console.warn("‚ö† Fullscreen nicht erlaubt! Wird automatisch deaktiviert.");
            exitFullscreen();
        }
    });


    /**
     * Verl√§sst den Fullscreen-Modus und stellt die Canvas-Gr√∂√üe zur√ºck.
     */
    function exitFullscreen() {
        if (document.fullscreenElement) {
            document.exitFullscreen().then(() => {
                restoreCanvasSize();
            }).catch(err => console.error("‚ùå Fehler beim Beenden von Fullscreen:", err));
        } else {
            console.warn("üì¢ Fullscreen war nicht aktiv.");
        }
    }

    /**
     * Stellt die Canvas-Gr√∂√üe auf Standard (720x480) zur√ºck.
     */
    function restoreCanvasSize() {
        canvas.style.removeProperty("width");
        canvas.style.removeProperty("height");
        canvas.style.position = "absolute";
        canvas.style.top = "50%";
        canvas.style.left = "50%";
        canvas.style.transform = "translate(-50%, -50%)";
        console.log("üîÑ Canvas-Gr√∂√üe auf 720x480 zur√ºckgesetzt.");
        ensureControlsVisible();
    }

    /**
     * √ñffnet oder schlie√üt das Settings-Men√º.
     */
    function toggleSettingsMenu() {
        if (!settingsMenu) {
            console.warn("‚ùå Element mit ID 'settings-menu' nicht gefunden.");
            return;
        }

        if (settingsMenu.style.display === "none" || settingsMenu.style.display === "") {

            settingsMenu.style.top = "50%";

            settingsMenu.style.color = "white";

            settingsMenu.style.borderRadius = "10px";
            settingsMenu.style.zIndex = "99999";
            console.log("‚öôÔ∏è Settings-Men√º ge√∂ffnet.");
        } else {
            settingsMenu.style.display = "none";
            console.log("‚öôÔ∏è Settings-Men√º geschlossen.");
        }
    }

    /**
     * Sorgt daf√ºr, dass die Steuerung IMMER sichtbar bleibt.
     */
    function ensureControlsVisible() {
        if (controlsVisible) {
            controls.style.display = "flex";
            touchControls.style.display = "flex";
        }
        showControlsButton.style.display = "flex"; // Der Button bleibt immer sichtbar
    }

    // Event-Listener f√ºr Buttons
    fullscreenButton.addEventListener("click", toggleFullscreen);
    exitButton.addEventListener("click", exitFullscreen);
    settingsButton.addEventListener("click", toggleSettingsMenu);

    // Pr√ºft, ob der Fullscreen-Modus verlassen wurde und stellt Buttons sicher
    document.addEventListener("fullscreenchange", () => {
        if (!document.fullscreenElement) {
            restoreCanvasSize();
        }
        ensureControlsVisible();
    });
});
