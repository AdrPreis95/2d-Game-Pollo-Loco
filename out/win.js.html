// Show the exit overlay with two options when the user clicks "Exit" or "Quit Game"

}

// Attempt to close the game window
function tryCloseWindow() {
    console.log("üõë Attempting to close the window...");
    window.close(); // Try closing the window

    // If the window is not closed, keep the overlay active
    setTimeout(() => {
        if (!window.closed) {
            console.warn("‚ùå The window could not be closed automatically.");
            document.getElementById("exit-overlay").innerHTML = `
                <p>‚ö† Your browser does not allow automatic closing.</p>
                <p>Please close the window manually or click "Go to about:blank".</p>
                <button onclick="redirectToBlank()">Go to about:blank</button>
            `;
        }
    }, 500);
}

// If the user chooses to be redirected instead
function redirectToBlank() {
    let newTab = window.open("", "_blank");

    if (newTab) {
        newTab.document.write(`
            <html>
            <head>
                <title>Game Closed</title>

                    body {
                        font-family: Arial, sans-serif;
                        text-align: center;

                    }

                        font-size: 20px;
                    }
                    button {
                        padding: 10px 20px;
                        font-size: 18px;
                        background-color: #ff3333;
                        color: white;
                        border: none;
                        cursor: pointer;
                        margin-top: 20px;
                    }
                    button:hover {
                        background-color: #cc0000;
                    }
                </style>
            </head>
            <body>
                <h1>‚ö† Game Closed</h1>
                <p>Your browser has blocked automatic closing.</p>
                <p>Please close this window manually.</p>
                <button onclick="window.close()">Got it</button>
            </body>
            </html>
        `);
        newTab.document.close();
    } else {
        console.error("‚ùå Popup blocker detected. No redirection will take place.");
    }
}
